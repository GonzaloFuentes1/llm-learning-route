End-to-end: generar N respuestas por prompt, puntuar con un RM, filtrar por calidad, iterar con el modelo mejorado. La receta usada en Phi-4, WizardLM y Nemotron.

El insight clave: el reward model actúa como filtro de calidad. En lugar de pedir a un humano que revise respuestas generadas, el RM puntúa cada una y descartamos las de baja calidad antes del fine-tuning. Con N respuestas por prompt podemos también construir pares (chosen, rejected) para DPO directamente.

TécnicaUsa RM paraSalida
Best-of-N filteringElegir la mejor respuesta de NDataset SFT limpio
Quality filteringDescartar respuestas con score bajoDataset SFT filtrado
Delta DPO / Online DPOConstruir pares chosen/rejectedDataset de preferencias
Curriculum learningOrdenar por dificultad (score)Secuencia de entrenamiento

3. Generar N respuestas por prompt

Generamos múltiples respuestas con temperatura alta para aumentar la diversidad. Con N=8 tenemos suficiente varianza para que el RM pueda discriminar.

python
from transformers import AutoModelForCausalLM, AutoTokenizer
import torch

model_name = "meta-llama/Llama-3.1-8B-Instruct"
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForCausalLM.from_pretrained(
    model_name, torch_dtype=torch.bfloat16, device_map="auto"
)
model.eval()

N = 8  # respuestas por prompt
TEMPERATURE = 0.9
MAX_NEW_TOKENS = 512

@torch.no_grad()
def generate_n_responses(prompt: str, n: int = N) -> list[str]:
    messages = [{"role": "user", "content": prompt}]
    input_ids = tokenizer.apply_chat_template(
        messages, add_generation_prompt=True, return_tensors="pt"
    ).to(model.device)

    outputs = model.generate(
        input_ids,
        max_new_tokens=MAX_NEW_TOKENS,
        do_sample=True,
        temperature=TEMPERATURE,
        top_p=0.95,
        num_return_sequences=n,       # ← genera N en paralelo
        pad_token_id=tokenizer.eos_token_id,
    )

    # Decodificar solo los tokens nuevos
    prompt_len = input_ids.shape[1]
    responses = []
    for out in outputs:
        text = tokenizer.decode(out[prompt_len:], skip_special_tokens=True)
        responses.append(text.strip())
    return responses
Batch por prompt: num_return_sequences=N genera N respuestas en un solo forward pass. Es mucho más eficiente que llamar generate() N veces.
4. Scoring con el Reward Model

Cargamos el RM entrenado (ver guía de Reward Models) y puntuamos cada respuesta. El score es un escalar: más alto = mejor.

python
from transformers import AutoModelForSequenceClassification
import torch

RM_PATH = "latamgpt/reward-model-v1"  # checkpoint del RM entrenado

rm_tokenizer = AutoTokenizer.from_pretrained(RM_PATH)
rm_model = AutoModelForSequenceClassification.from_pretrained(
    RM_PATH, num_labels=1, torch_dtype=torch.bfloat16, device_map="cuda"
)
rm_model.eval()

@torch.no_grad()
def score_response(prompt: str, response: str) -> float:
    """Devuelve el reward scalar para un par (prompt, response)."""
    messages = [
        {"role": "user", "content": prompt},
        {"role": "assistant", "content": response},
    ]
    text = rm_tokenizer.apply_chat_template(messages, tokenize=False)
    enc = rm_tokenizer(text, return_tensors="pt", truncation=True, max_length=2048)
    enc = {k: v.to(rm_model.device) for k, v in enc.items()}
    logits = rm_model(**enc).logits
    return logits[0, 0].item()

@torch.no_grad()
def score_all(prompt: str, responses: list[str]) -> list[float]:
    return [score_response(prompt, r) for r in responses]
Scoring en batch con vLLM (producción)

Para escalar a millones de prompts, es más eficiente usar vLLM para la generación y el RM como un servicio separado.

python
from vllm import LLM, SamplingParams
import torch

    tokenizer.apply_chat_template(
        [{"role": "user", "content": p}],
        add_generation_prompt=True, tokenize=False
    )
    for p in seed_prompts
]

outputs = llm.generate(prompts_text, sampling)

5. Filtrado por calidad y construcción del dataset

Usamos el score del RM para filtrar respuestas de baja calidad. El umbral óptimo depende de la distribución de scores — típicamente mantenemos el top 50-70% de respuestas.

python
from datasets import Dataset

QUALITY_THRESHOLD = 0.0   # score mínimo (normalizado según tu RM)
KEEP_TOP_K = 1            # para best-of-N SFT, guardar solo la mejor

def build_sft_dataset(seed_prompts: list[str]) -> Dataset:
    rows = []
    for prompt in seed_prompts:
        responses = generate_n_responses(prompt, n=N)
        scores   = score_all(prompt, responses)

        # Ordenar por score descendente
        ranked = sorted(zip(scores, responses), reverse=True)

        # Estrategia A: best-of-N (solo la mejor para SFT)
        best_score, best_response = ranked[0]
        if best_score >= QUALITY_THRESHOLD:
            rows.append({
                "messages": [
                    {"role": "user",      "content": prompt},
                    {"role": "assistant", "content": best_response},
                ],
                "rm_score": best_score,
                "language": "es",
            })

    return Dataset.from_list(rows)


def build_dpo_dataset(seed_prompts: list[str], delta: float = 1.5) -> Dataset:
    """Construye pares (chosen, rejected) con delta filtering (receta OLMo 3)."""
    rows = []
    for prompt in seed_prompts:
        responses = generate_n_responses(prompt, n=N)
        scores   = score_all(prompt, responses)

        ranked = sorted(zip(scores, responses), reverse=True)
        best_score, best_resp   = ranked[0]
        worst_score, worst_resp = ranked[-1]

        # Solo incluir pares donde la diferencia es significativa
        if best_score - worst_score >= delta:
            rows.append({
                "prompt":   [{"role": "user", "content": prompt}],
                "chosen":   [{"role": "assistant", "content": best_resp}],
                "rejected": [{"role": "assistant", "content": worst_resp}],
                "score_chosen":   best_score,
                "score_rejected": worst_score,
            })

    return Dataset.from_list(rows)
Delta filtering: Solo incluir pares donde score_chosen − score_rejected ≥ δ asegura que el RM tiene una opinión clara. Pares con delta pequeño son ruidosos y pueden confundir al DPO trainer.
6. Curriculum learning

En lugar de entrenar con todos los datos en orden aleatorio, ordenamos por dificultad: primero las instrucciones más fáciles (que el modelo ya puede responder bien) y progresamos hacia las difíciles. El RM score de la mejor respuesta aproxima la "facilidad".

python
from datasets import Dataset

def apply_curriculum(ds: Dataset, strategy: str = "easy_first") -> Dataset:
    """
    easy_first:  rm_score alto primero (modelo ya lo maneja bien)
    hard_first:  rm_score bajo primero (fuerza al modelo a aprender cosas difíciles)
    mixed:       intercalar fácil y difícil
    """
    scores = ds["rm_score"]
    indices = sorted(range(len(scores)), key=lambda i: scores[i],
                     reverse=(strategy == "easy_first"))
    return ds.select(indices)

# Combinar SFT datasets de múltiples iteraciones con pesos
from datasets import concatenate_datasets

7. Pipeline iterativo completo

El poder real viene de iterar: el modelo mejorado genera mejores respuestas, que a su vez producen mejores datos de entrenamiento para la siguiente ronda.

python
from pathlib import Path
from trl import SFTTrainer, SFTConfig
from transformers import AutoModelForCausalLM, AutoTokenizer

def run_pipeline_iteration(
    model_path: str,
    seed_prompts: list[str],
    iteration: int,
    output_dir: str = "output",
) -> str:
    """Ejecuta una iteración: genera datos → filtra → entrena → guarda."""
    print(f"\n=== Iteración {iteration} ===")

    # 1. Generar y filtrar datos con el modelo actual
    print("Generando respuestas...")
    sft_ds = build_sft_dataset(seed_prompts)
    print(f"  Dataset: {len(sft_ds)} ejemplos (rm_score >= {QUALITY_THRESHOLD})")

    sft_ds.save_to_disk(f"{output_dir}/iter{iteration}_data")

    # 2. Fine-tuning SFT
    print("Entrenando...")
    model = AutoModelForCausalLM.from_pretrained(
        model_path, torch_dtype=torch.bfloat16, device_map="auto"
    )
    tokenizer = AutoTokenizer.from_pretrained(model_path)

    trainer = SFTTrainer(
        model=model,
        tokenizer=tokenizer,
        train_dataset=sft_ds,
        args=SFTConfig(
            output_dir=f"{output_dir}/iter{iteration}_ckpt",
            num_train_epochs=2,
            per_device_train_batch_size=4,
            gradient_accumulation_steps=4,
            learning_rate=2e-5,
            lr_scheduler_type="cosine",
            warmup_ratio=0.05,
            bf16=True,
            save_strategy="epoch",
            logging_steps=50,
        ),
    )
    trainer.train()

    new_model_path = f"{output_dir}/iter{iteration}_model"
    trainer.save_model(new_model_path)
    print(f"  Modelo guardado en: {new_model_path}")
    return new_model_path


# Loop principal
BASE_MODEL = "meta-llama/Llama-3.1-8B-Instruct"
N_ITERATIONS = 3

current_model = BASE_MODEL
for i in range(1, N_ITERATIONS + 1):
    current_model = run_pipeline_iteration(
        model_path=current_model,
        seed_prompts=seed_prompts,
        iteration=i,
    )
8. Detección de model collapse

El model collapse ocurre cuando los datos sintéticos de las rondas posteriores son menos diversos que los de la ronda 1, eventualmente colapsando a respuestas repetitivas. Hay que monitorearlo explícitamente.

python
import numpy as np
from collections import Counter

def measure_diversity(responses: list[str]) -> dict:
    """Métricas de diversidad para detectar model collapse."""

    # 1. Distinct-n: fracción de n-gramas únicos
    def distinct_n(texts, n):
        all_ngrams = []
        for text in texts:
            tokens = text.split()
            all_ngrams.extend(zip(*[tokens[i:] for i in range(n)]))
        if not all_ngrams:
            return 0.0
        return len(set(all_ngrams)) / len(all_ngrams)

    # 2. Varianza del score del RM (debe mantenerse alta)
    # (pasarías los scores como argumento adicional)

    # 3. Longitud media y varianza
    lengths = [len(r.split()) for r in responses]

    return {
        "distinct_1": distinct_n(responses, 1),
        "distinct_2": distinct_n(responses, 2),
        "mean_length": np.mean(lengths),
        "std_length":  np.std(lengths),
        "n_unique_responses": len(set(responses)),
    }

# Comparar entre iteraciones
for i, iter_ds in enumerate(iter_datasets, 1):
    responses = [ex["messages"][1]["content"] for ex in iter_ds]
    metrics = measure_diversity(responses)
    print(f"Iter {i}: distinct-2={metrics['distinct_2']:.3f}, "
          f"len_std={metrics['std_length']:.1f}")
Señal de alertaQué indicaMitigación
distinct-2 < 0.3Respuestas muy repetitivasAumentar temperatura, añadir prompts seed humanos
std_length → 0Respuestas de longitud uniformeAñadir ejemplos cortos y largos explícitamente

Regla práctica: Siempre mezclar entre 5-10% de los datos originales (de ronda 0 o datos humanos) en cada iteración. Esto ancla el modelo y previene drift acumulativo. Sin esto, después de 3-4 rondas el modelo puede degradar en tareas que no están en los prompts semilla.
