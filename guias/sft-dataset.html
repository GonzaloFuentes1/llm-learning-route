Construcción de datasets SFT para Latam

Cómo crear datasets de instrucciones de alta calidad en español y portugués latinoamericano. Cubre formatos de datos, generación sintética con Self-Instruct y Magpie, traducción verificada, filtrado y publicación en HuggingFace Hub.

1. Formatos de datos SFT

El estándar actual es el formato conversacional (lista de mensajes). Cada modelo tiene su propio chat template; siempre usar apply_chat_template en lugar de construir el string manualmente.

json — formato conversacional (recomendado)
{
  "messages": [
    {"role": "system",    "content": "Eres un asistente experto en legislación latinoamericana."},
    {"role": "user",      "content": "¿Cuál es la diferencia entre una SRL y una SA en Argentina?"},
    {"role": "assistant", "content": "En Argentina, las principales diferencias son..."}
  ],
  "source": "synthetic_selfinstruct",
  "language": "es-AR",
  "quality_score": 0.87
}
CampoRequeridoDescripción
messagesSíLista ordenada de turnos. role ∈ {system, user, assistant}
sourceRecomendadoOrigen del ejemplo (permite filtrar y auditar)
languageRecomendadoBCP-47: es-MX, es-AR, pt-BR, etc.
quality_scoreOpcionalScore de RM o clasificador, útil para filtrado posterior

2. Generación con Self-Instruct

Self-Instruct parte de un conjunto de instrucciones semilla (seed tasks) escritas por humanos y las usa para prompting un LLM que genera nuevas instrucciones + respuestas.

python — self_instruct.py
import json
import random
from openai import OpenAI  # compatible con cualquier API OpenAI-compatible

client = OpenAI(
    api_key="tu-api-key",
    # base_url="http://localhost:8000/v1",  # usar modelo local con vLLM
)

SEED_TASKS = [
    "Explica qué es el FONASA y cómo acceder a él.",
    "¿Cómo calculo el aguinaldo en México?",
    "Escribe un correo formal solicitando vacaciones.",
    "¿Cuáles son los derechos del consumidor en Brasil?",
    "Traduce este párrafo al portugués brasileño conservando el registro formal.",
]

SYSTEM_PROMPT = """Eres un experto generando instrucciones de entrenamiento en español e
inglés latinoamericano. Genera 5 instrucciones nuevas, diversas y útiles para un asistente
de IA. Las instrucciones deben ser en español rioplatense, mexicano, colombiano, chileno o
portugués brasileño. Responde solo con JSON, sin texto adicional."""

def generate_instructions(seed_sample: list[str]) -> list[dict]:
    seed_str = "\n".join(f"- {s}" for s in seed_sample)
    prompt = f"Instrucciones de ejemplo:\n{seed_str}\n\nGenera 5 instrucciones nuevas en JSON:"

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user",   "content": prompt},
        ],
        response_format={"type": "json_object"},
        temperature=0.9,
    )
    data = json.loads(response.choices[0].message.content)
    return data.get("instructions", [])


def generate_response(instruction: str) -> str:
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "system",    "content": "Eres un asistente útil, preciso y culturalmente apropiado para Latinoamérica."},
            {"role": "user",      "content": instruction},
        ],
        temperature=0.7,
        max_tokens=1024,
    )
    return response.choices[0].message.content


# Pipeline principal
all_examples = []
seeds = SEED_TASKS.copy()

for iteration in range(50):  # genera 250 instrucciones
    sample = random.sample(seeds, min(5, len(seeds)))
    new_instructions = generate_instructions(sample)
    for instr in new_instructions:
        if isinstance(instr, str) and len(instr) > 20:
            response = generate_response(instr)
            example = {
                "messages": [
                    {"role": "user",      "content": instr},
                    {"role": "assistant", "content": response},
                ],
                "source": "self_instruct",
            }
            all_examples.append(example)
            seeds.append(instr)   # agregar al pool de seeds

# Guardar
with open("data/self_instruct_es.jsonl", "w", encoding="utf-8") as f:
    for ex in all_examples:
        f.write(json.dumps(ex, ensure_ascii=False) + "\n")

print(f"Generados {len(all_examples)} ejemplos")

3. Magpie: generación sin seed tasks

Magpie extrae instrucciones directamente del modelo usando su sistema de templates. No necesita seed tasks: el modelo genera la pregunta del usuario a partir del prompt de sistema truncado.

python — magpie_es.py
from transformers import AutoTokenizer, AutoModelForCausalLM
import torch
import json

MODEL = "meta-llama/Llama-3.2-8B-Instruct"
tokenizer = AutoTokenizer.from_pretrained(MODEL)
model = AutoModelForCausalLM.from_pretrained(MODEL, torch_dtype=torch.bfloat16, device_map="auto")


def extract_user_turn(system_prompt: str = None) -> str:
    """Genera una instrucción de usuario usando el chat template truncado."""
    messages = []
    if system_prompt:
        messages.append({"role": "system", "content": system_prompt})
    # Aplicar template hasta el punto donde empieza el turno del usuario
    prompt = tokenizer.apply_chat_template(
        messages,
        tokenize=False,
        add_generation_prompt=True,
    )
    # Para Llama3: el template termina con "<|start_header_id|>user<|end_header_id|>\n\n"
    # El modelo completará con la instrucción del usuario
    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)
    with torch.no_grad():
        output = model.generate(
            **inputs,
            max_new_tokens=256,
            temperature=1.0,
            do_sample=True,
            eos_token_id=tokenizer.convert_tokens_to_ids("<|eot_id|>"),
        )
    generated = tokenizer.decode(output[0][inputs.input_ids.shape[1]:], skip_special_tokens=True)
    return generated.strip()


def generate_response(instruction: str, system: str = None) -> str:
    messages = []
    if system:
        messages.append({"role": "system", "content": system})
    messages.append({"role": "user", "content": instruction})
    prompt = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)
    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)
    with torch.no_grad():
        output = model.generate(**inputs, max_new_tokens=1024, temperature=0.7, do_sample=True)
    return tokenizer.decode(output[0][inputs.input_ids.shape[1]:], skip_special_tokens=True).strip()


SYSTEM_ES = "Eres un asistente de IA útil, amable y preciso que responde siempre en español latino."

examples = []
for i in range(500):
    instruction = extract_user_turn(system_prompt=SYSTEM_ES)
    if len(instruction) < 15:
        continue
    response = generate_response(instruction, system=SYSTEM_ES)
    examples.append({
        "messages": [
            {"role": "system", "content": SYSTEM_ES},
            {"role": "user", "content": instruction},
            {"role": "assistant", "content": response},
        ],
        "source": "magpie",
    })
    if i % 50 == 0:
        print(f"[{i}/500] Ejemplo: {instruction[:80]}...")

with open("data/magpie_es.jsonl", "w", encoding="utf-8") as f:
    for ex in examples:
        f.write(json.dumps(ex, ensure_ascii=False) + "\n")

4. Filtrado de calidad

python — filter_sft.py
import json
import fasttext
from pathlib import Path

# Cargar detector de idioma
lid_model = fasttext.load_model("lid.176.bin")

def detect_language(text: str) -> tuple[str, float]:
    labels, probs = lid_model.predict(text.replace("\n", " "), k=1)
    lang = labels[0].replace("__label__", "")
    return lang, float(probs[0])


def is_valid(example: dict, target_langs=("es", "pt")) -> bool:
    msgs = example.get("messages", [])
    if len(msgs) < 2:
        return False

    # Filtro de longitud
    user_content = next((m["content"] for m in msgs if m["role"] == "user"), "")
    asst_content = next((m["content"] for m in msgs if m["role"] == "assistant"), "")
    if len(user_content) < 10 or len(asst_content) < 20:
        return False
    if len(asst_content) > 4000:  # respuestas excesivamente largas
        return False

    # Filtro de idioma
    lang, conf = detect_language(asst_content)
    if lang not in target_langs or conf < 0.85:
        return False

    # Filtro de contenido (básico)
    bad_phrases = ["as an AI language model", "I cannot", "I'm sorry, I can't"]
    if any(p in asst_content for p in bad_phrases):
        return False

    return True


input_files = ["data/self_instruct_es.jsonl", "data/magpie_es.jsonl"]
output_file  = "data/sft_latam_filtered.jsonl"

total, kept = 0, 0
with open(output_file, "w", encoding="utf-8") as out:
    for path in input_files:
        for line in Path(path).read_text(encoding="utf-8").splitlines():
            example = json.loads(line)
            total += 1
            if is_valid(example):
                out.write(json.dumps(example, ensure_ascii=False) + "\n")
                kept += 1

print(f"Filtrado: {kept}/{total} ejemplos ({100*kept/total:.1f}%)")

5. Publicar en HuggingFace Hub

python — upload_dataset.py
from datasets import load_dataset, DatasetDict
from huggingface_hub import HfApi

# Cargar desde JSONL local
dataset = load_dataset("json", data_files={
    "train": "data/sft_latam_filtered.jsonl",
})

# Split train/validation
split = dataset["train"].train_test_split(test_size=0.05, seed=42)
dataset = DatasetDict({"train": split["train"], "validation": split["test"]})

# Subir
dataset.push_to_hub(
    "latamgpt/sft-latam-v1",
    private=True,
    commit_message="SFT dataset v1 — es/pt Latam, 50K ejemplos filtrados",
)
print("Dataset subido a HuggingFace Hub")
