Anonimización y PII

Detectar y reemplazar información de identificación personal (PII) en corpus de texto antes de entrenamiento. Cubre Microsoft Presidio con soporte para español/portugués y formatos latinoamericanos.

1. Tipos de PII relevantes para Latam

Los modelos que aprenden de texto con PII pueden "memorizar" y reproducir datos personales. La detección y remoción es un requisito legal (LGPD en Brasil, Ley 19.628 en Chile) y una buena práctica de seguridad.

text — categorías de PII
PII universal:
  - Nombres de personas (PERSON)
  - Emails (EMAIL_ADDRESS)
  - Números de teléfono (PHONE_NUMBER)
  - Direcciones IP (IP_ADDRESS)
  - Fechas de nacimiento (DATE_TIME + contexto)
  - Números de tarjeta de crédito (CREDIT_CARD)
  - URLs personales

PII específica de Latam:
  - RUT (Chile):        12.345.678-9
  - CUIT/CUIL (Arg):   20-12345678-9
  - CPF (Brasil):       123.456.789-00
  - DNI (España/Arg):   12345678A
  - Cédula (Colombia):  1.234.567.890
  - Pasaportes

2. Pipeline con Microsoft Presidio

bash
pip install presidio-analyzer presidio-anonymizer spacy
python -m spacy download es_core_news_lg    # NER en español
python -m spacy download pt_core_news_lg    # NER en portugués
python — presidio_pipeline.py
from presidio_analyzer import AnalyzerEngine
from presidio_analyzer.nlp_engine import NlpEngineProvider
from presidio_anonymizer import AnonymizerEngine
from presidio_anonymizer.entities import OperatorConfig

# Configurar motor NLP con spaCy español
configuration = {
    "nlp_engine_name": "spacy",
    "models": [
        {"lang_code": "es", "model_name": "es_core_news_lg"},
        {"lang_code": "pt", "model_name": "pt_core_news_lg"},
    ],
}
provider = NlpEngineProvider(nlp_configuration=configuration)
nlp_engine = provider.create_engine()

analyzer = AnalyzerEngine(nlp_engine=nlp_engine, supported_languages=["es", "pt"])
anonymizer = AnonymizerEngine()

# Operadores: reemplazar con tokens genéricos
operators = {
    "PERSON":        OperatorConfig("replace", {"new_value": "<PERSONA>"}),
    "EMAIL_ADDRESS": OperatorConfig("replace", {"new_value": "<EMAIL>"}),
    "PHONE_NUMBER":  OperatorConfig("replace", {"new_value": "<TELEFONO>"}),
    "IP_ADDRESS":    OperatorConfig("replace", {"new_value": "<IP>"}),
    "CREDIT_CARD":   OperatorConfig("replace", {"new_value": "<TARJETA>"}),
    "URL":           OperatorConfig("replace", {"new_value": "<URL>"}),
    "LOCATION":      OperatorConfig("keep"),  # mantener ciudades (útil para Latam)
}


def anonymize_text(text: str, lang: str = "es") -> str:
    """Anonimiza PII en un texto. Retorna el texto con tokens de reemplazo."""
    results = analyzer.analyze(text=text, language=lang)
    if not results:
        return text
    anonymized = anonymizer.anonymize(
        text=text,
        analyzer_results=results,
        operators=operators,
    )
    return anonymized.text


# Ejemplo
texto = "El Dr. Juan García (juan.garcia@hospital.cl, +56 9 1234 5678) vive en Santiago."
print(anonymize_text(texto))
# Output: "El Dr. <PERSONA> (<EMAIL>, <TELEFONO>) vive en Santiago."

3. Reconocedores custom para formatos latinoamericanos

python — recognizers_latam.py
import re
from presidio_analyzer import PatternRecognizer, Pattern


# RUT chileno: 12.345.678-9 o 12345678-9
rut_pattern = Pattern(
    name="rut_pattern",
    regex=r"\b\d{1,2}\.?\d{3}\.?\d{3}-[\dkK]\b",
    score=0.85,
)
rut_recognizer = PatternRecognizer(
    supported_entity="RUT_CL",
    patterns=[rut_pattern],
    context=["rut", "run", "cédula", "documento"],
)

# CUIT/CUIL argentino: 20-12345678-9
cuit_pattern = Pattern(
    name="cuit_pattern",
    regex=r"\b(20|23|24|27|30|33|34)-\d{8}-[0-9]\b",
    score=0.85,
)
cuit_recognizer = PatternRecognizer(
    supported_entity="CUIT_AR",
    patterns=[cuit_pattern],
    context=["cuit", "cuil", "afip"],
)

# CPF brasileño: 123.456.789-00
cpf_pattern = Pattern(
    name="cpf_pattern",
    regex=r"\b\d{3}\.?\d{3}\.?\d{3}-?\d{2}\b",
    score=0.80,
)
cpf_recognizer = PatternRecognizer(
    supported_entity="CPF_BR",
    patterns=[cpf_pattern],
    context=["cpf", "cadastro"],
)


# Agregar reconocedores al analyzer
def build_latam_analyzer():
    from presidio_analyzer import AnalyzerEngine
    from presidio_analyzer.nlp_engine import NlpEngineProvider

    provider = NlpEngineProvider(nlp_configuration={
        "nlp_engine_name": "spacy",
        "models": [{"lang_code": "es", "model_name": "es_core_news_lg"}],
    })
    engine = provider.create_engine()
    analyzer = AnalyzerEngine(nlp_engine=engine, supported_languages=["es"])

    # Registrar reconocedores custom
    analyzer.registry.add_recognizer(rut_recognizer)
    analyzer.registry.add_recognizer(cuit_recognizer)
    analyzer.registry.add_recognizer(cpf_recognizer)

    return analyzer

4. Procesar corpus a escala

python — anonimizar_corpus.py
import json
from multiprocessing import Pool
from pathlib import Path
from presidio_anonymizer import AnonymizerEngine
from presidio_anonymizer.entities import OperatorConfig

anonymizer = AnonymizerEngine()
analyzer = None  # se inicializa en cada worker

def init_worker():
    global analyzer
    analyzer = build_latam_analyzer()  # función del paso anterior

def process_document(doc: dict) -> dict:
    lang = doc.get("lang", "es")
    try:
        results = analyzer.analyze(text=doc["text"], language=lang)
        if results:
            anon = anonymizer.anonymize(
                text=doc["text"],
                analyzer_results=results,
                operators={
                    "PERSON": OperatorConfig("replace", {"new_value": "<PERSONA>"}),
                    "EMAIL_ADDRESS": OperatorConfig("replace", {"new_value": "<EMAIL>"}),
                    "PHONE_NUMBER": OperatorConfig("replace", {"new_value": "<TELEFONO>"}),
                    "IP_ADDRESS": OperatorConfig("replace", {"new_value": "<IP>"}),
                    "RUT_CL": OperatorConfig("replace", {"new_value": "<ID_FISCAL>"}),
                    "CUIT_AR": OperatorConfig("replace", {"new_value": "<ID_FISCAL>"}),
                    "CPF_BR": OperatorConfig("replace", {"new_value": "<ID_FISCAL>"}),
                },
            )
            return {**doc, "text": anon.text, "pii_detected": len(results)}
    except Exception:
        pass
    return {**doc, "pii_detected": 0}


def anonymize_corpus(input_file: str, output_file: str, workers: int = 16):
    docs = [json.loads(line) for line in open(input_file)]

    with Pool(workers, initializer=init_worker) as pool:
        processed = pool.map(process_document, docs)

    docs_with_pii = sum(1 for d in processed if d.get("pii_detected", 0) > 0)
    print(f"Documentos con PII: {docs_with_pii:,} / {len(docs):,} ({docs_with_pii/len(docs):.1%})")

    with open(output_file, "w") as f:
        for doc in processed:
            f.write(json.dumps(doc, ensure_ascii=False) + "\n")
Limitación importante: Presidio + spaCy no detecta el 100% del PII. Especialmente nombres propios en contextos inusuales o documentos de baja calidad pueden escapar. Valida siempre muestreando manualmente 200-500 documentos post-procesados y ajusta los thresholds según el caso de uso.
