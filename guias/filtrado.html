Filtrado de Calidad de Datos

Pipeline completo para filtrar documentos de baja calidad de un corpus web: reglas heurísticas, clasificador de idioma fastText, clasificador de calidad fastText y filtro de perplejidad KenLM.

1. Por qué el filtrado importa

DCLM (2024) demostró experimentalmente que la calidad del pipeline de datos supera a la arquitectura del modelo: un modelo más pequeño entrenado con datos bien filtrados supera a uno más grande con datos crudos. El filtrado es posiblemente el componente con mayor ROI en el pipeline.

Flujo típico: corpus crudo → filtro de idioma → reglas Gopher/C4 → clasificador de calidad → filtro de perplejidad → corpus limpio para training

2. Reglas heurísticas (Gopher / C4)

Filtros rápidos basados en estadísticas del texto. Eliminan spam, listas de palabras, contenido autogenerado y páginas rotas.

python — reglas_heuristicas.py
import re
import unicodedata


def passes_gopher_rules(text: str) -> bool:
    """
    Reglas inspiradas en el paper de Gopher (DeepMind, 2022).
    Returns True si el documento pasa el filtro.
    """
    words = text.split()
    n_words = len(words)
    n_chars = len(text)

    # 1. Longitud mínima y máxima
    if not (50 <= n_words <= 100_000):
        return False

    # 2. Ratio de caracteres alfabéticos
    alpha_chars = sum(1 for c in text if c.isalpha())
    if alpha_chars / n_chars < 0.70:
        return False

    # 3. Palabras promedio no demasiado largas (spam de SEO)
    avg_word_len = sum(len(w) for w in words) / n_words
    if not (3 <= avg_word_len <= 10):
        return False

    # 4. No más del 10% de líneas terminan con "..."
    lines = text.split("\n")
    ellipsis_lines = sum(1 for l in lines if l.rstrip().endswith("..."))
    if len(lines) > 0 and ellipsis_lines / len(lines) > 0.10:
        return False

    # 5. No más del 30% de líneas son duplicadas (listas repetitivas)
    non_empty_lines = [l for l in lines if l.strip()]
    if len(non_empty_lines) > 5:
        unique_ratio = len(set(non_empty_lines)) / len(non_empty_lines)
        if unique_ratio < 0.70:
            return False

    # 6. Ratio de símbolos (filtrar código o contenido técnico irrelevante)
    symbol_ratio = sum(1 for c in text if c in "{}[]<>|#@") / n_chars
    if symbol_ratio > 0.05:
        return False

    return True


# Aplicar a dataset
def filter_corpus(input_path: str, output_path: str):
    kept, removed = 0, 0
    with open(input_path) as fin, open(output_path, "w") as fout:
        for line in fin:
            import json
            doc = json.loads(line)
            if passes_gopher_rules(doc["text"]):
                fout.write(line)
                kept += 1
            else:
                removed += 1
    print(f"Conservados: {kept:,} ({kept/(kept+removed):.1%}) | Eliminados: {removed:,}")

python — deteccion_idioma.py
import fasttext

lid_model = fasttext.load_model("lid.176.bin")

def detect_language(text: str, threshold: float = 0.65) -> tuple[str, float]:
    """Detecta el idioma de un texto. Retorna (lang_code, confidence)."""
    # fastText requiere texto sin saltos de línea
    clean = text.replace("\n", " ").strip()[:500]  # solo primeros 500 chars
    predictions = lid_model.predict(clean, k=1)
    lang = predictions[0][0].replace("__label__", "")
    confidence = predictions[1][0]
    return lang, confidence


def is_spanish_or_portuguese(text: str, threshold: float = 0.65) -> bool:
    lang, conf = detect_language(text)
    return lang in ("es", "pt") and conf >= threshold


# Filtrar corpus por idioma
def filter_by_language(docs: list[dict], target_langs=("es", "pt"), threshold=0.65):
    return [
        doc for doc in docs
        if detect_language(doc["text"])[0] in target_langs
        and detect_language(doc["text"])[1] >= threshold
    ]

4. Clasificador de calidad fastText

Entrenamos un clasificador binario: Wikipedia/textos de calidad como clase positiva vs. web crawl crudo como clase negativa. Permite filtrar documentos que "se parecen a" texto de referencia.

python — entrenar_clasificador_calidad.py
import fasttext
from pathlib import Path

# Preparar datos de entrenamiento en formato fastText
# Positivos: Wikipedia en español/portugués
# Negativos: muestra del web crawl crudo

def prepare_training_file(
    positives: list[str],   # textos de Wikipedia
    negatives: list[str],   # textos de web crawl
    output_path: str,
    max_chars: int = 300,
):
    with open(output_path, "w") as f:
        for text in positives:
            clean = text.replace("\n", " ")[:max_chars]
            f.write(f"__label__quality {clean}\n")
        for text in negatives:
            clean = text.replace("\n", " ")[:max_chars]
            f.write(f"__label__low_quality {clean}\n")


# Entrenar el modelo
def train_quality_classifier(train_file: str, val_file: str) -> fasttext.FastText._FastText:
    model = fasttext.train_supervised(
        input=train_file,
        epoch=3,
        lr=0.1,
        wordNgrams=2,        # bigramas para mejor contexto
        dim=256,
        loss="hs",           # hierarchical softmax (más rápido)
        thread=16,
    )
    result = model.test(val_file)
    print(f"N={result[0]:,}, P@1={result[1]:.3f}, R@1={result[2]:.3f}")
    return model


# Usar el clasificador
def score_document(model, text: str) -> float:
    """Retorna probabilidad de que el documento sea de alta calidad."""
    clean = text.replace("\n", " ")[:300]
    pred = model.predict(clean, k=2)
    labels, probs = pred
    quality_idx = labels.index("__label__quality")
    return probs[quality_idx]


# Filtrar corpus
model = fasttext.load_model("quality_classifier_es.bin")
HIGH_QUALITY_THRESHOLD = 0.5

filtered = [doc for doc in corpus if score_document(model, doc["text"]) >= HIGH_QUALITY_THRESHOLD]
print(f"Conservados: {len(filtered):,} / {len(corpus):,} ({len(filtered)/len(corpus):.1%})")

5. Filtro de perplejidad con KenLM

Documentos con perplejidad muy alta contienen texto incoherente o generado automáticamente. Un modelo de lenguaje de n-gramas entrenado en Wikipedia permite detectarlos eficientemente.

bash — instalar KenLM
pip install kenlm

# Descargar modelo KenLM en español (5-gramas, entrenado en Wikipedia)
# Se puede entrenar propio o usar uno pre-entrenado de CCNet:
wget https://dl.fbaipublicfiles.com/cc_net/lm/es.arpa.bin
python — filtro_perplejidad.py
import kenlm
import math

model = kenlm.Model("es.arpa.bin")

def compute_perplexity(text: str, model: kenlm.Model) -> float:
    """Calcula la perplejidad de un texto con el modelo KenLM."""
    words = text.lower().split()
    if len(words) < 10:
        return float("inf")

    log_prob = model.score(text.lower())  # log base 10
    n_words = len(list(model.full_scores(text.lower())))
    return 10 ** (-log_prob / n_words)


# Calibrar umbrales en tu corpus
# Típicamente: PPL < 100 → texto limpio, PPL > 500 → spam/incoherente
def filter_by_perplexity(
    docs: list[dict],
    min_ppl: float = 10,
    max_ppl: float = 500,
) -> list[dict]:
    filtered = []
    for doc in docs:
        ppl = compute_perplexity(doc["text"], model)
        if min_ppl <= ppl <= max_ppl:
            filtered.append({**doc, "perplexity": ppl})
    return filtered

6. Pipeline completo con Datatrove

python — pipeline_filtrado.py
from datatrove.executor import LocalPipelineExecutor
from datatrove.pipeline.readers import JsonlReader
from datatrove.pipeline.writers import JsonlWriter
from datatrove.pipeline.filters import (
    GopherQualityFilter,
    GopherRepetitionFilter,
    LanguageFilter,
    C4QualityFilter,
)
from datatrove.pipeline.dedup import MinhashDedupSignature

executor = LocalPipelineExecutor(
    pipeline=[
        JsonlReader(data_folder="data/raw/", glob_pattern="**/*.jsonl"),

        # 1. Filtro de idioma (conservar es y pt)
        LanguageFilter(languages=["es", "pt"], language_threshold=0.65),

        # 2. Reglas Gopher de calidad básica
        GopherQualityFilter(
            min_doc_words=50,
            max_doc_words=100_000,
            min_avg_word_length=3,
            max_avg_word_length=10,
            max_symbol_word_ratio=0.1,
        ),

        # 3. Filtro de repetición
        GopherRepetitionFilter(
            max_duplicated_5_grams_char_fraction=0.15,
            max_top_ngram_character_fraction=0.20,
        ),

        # 4. Filtros C4 adicionales
        C4QualityFilter(
            filter_no_terminal_punct=True,
            min_num_sentences=3,
        ),

        # Guardar corpus filtrado
        JsonlWriter(output_folder="data/filtered/"),
    ],
    tasks=64,
    workers=16,
    logging_dir="logs/filtering/",
)

executor.run()
Orden importa: Poner los filtros más baratos primero (longitud, idioma) elimina documentos antes de aplicar filtros costosos (perplejidad, clasificadores ML). Esto reduce el tiempo total de procesamiento.

7. DCLM: clasificador OH2.5 + ELI5 vs RefinedWeb

La innovación principal de DCLM fue usar datos de instrucción (OpenHermes 2.5 + ELI5 Reddit) como positivos en vez de Wikipedia. La hipótesis: si quieres que el modelo escriba como en instrucciones de alta calidad, entrena el clasificador con esos ejemplos. El umbral real usado en DCLM-Baseline es 0.018112 — muy permisivo, deja pasar casi todo excepto spam claro.

python — alternativa directa con fasttext
import fasttext

model = fasttext.load_model("fasttext_oh_eli5.bin")

def dclm_score(text: str) -> float:
    """
    Retorna P(hq) según el clasificador DCLM.
    Threshold del paper: 0.018112 (muy permisivo, elimina solo spam claro).
    """
    clean = text.replace("\n", " ").strip()[:300]
    pred = model.predict(clean, k=2)
    labels, probs = pred
    # Si el label "hq" es primero, su probabilidad es probs[0]
    if labels[0] == "__label__hq":
        return float(probs[0])
    return float(1.0 - probs[0])


# Filtrar corpus
DCLM_THRESHOLD = 0.018112  # del paper; sube a 0.1-0.3 para corpus más pequeño/limpio
filtered = [doc for doc in corpus if dclm_score(doc["text"]) >= DCLM_THRESHOLD]
OH2.5 vs Wikipedia como positivos: Wikipedia captura "texto enciclopédico bien escrito". OpenHermes/ELI5 captura "texto explicativo, instructivo, útil para un asistente". Para un LLM de chat/instrucción, el segundo da mejores resultados downstream según los ablaciones de DCLM.

8. Dolma Toolkit — pipeline de OLMo/AllenAI

Dolma es el toolkit de curaciónde datos de AllenAI, usado para construir el corpus de OLMo. Está escrito en Rust + Python, es muy rápido y usa deduplicación con Bloom filters en Rust. Los "taggers" son módulos Python que anotantributos sobre cada documento; el paso de mezcla/filtrado aplica los umbrales.

bash — instalar y aplicar taggers
pip install dolma

# Ver todos los taggers disponibles
dolma list

# Taggers clave disponibles:
# gopher_quality_v1          → reglas de calidad de Gopher (longitud, símbolos, etc.)
# gopher_repetition_v1       → detección de repeticiones n-gram
# c4_quality_v1              → reglas del dataset C4 (puntuación terminal, etc.)
# ft_lang_id_en_doc_v2       → detección de idioma con fastText
# jigsaw_hatespeech_v2       → contenido de odio (clasificador Jigsaw)
# pii_presidio_v1            → detección de PII con Presidio

# Aplicar taggers sobre el corpus (anota atributos, no filtra todavía)
dolma tag \
  --taggers gopher_quality_v1 c4_quality_v1 ft_lang_id_en_doc_v2 \
  --documents "data/raw/*.jsonl.gz" \
  --work_dir data/work/ \
  --processes 16
python — filtrar con los atributos de Dolma
from dolma.core.data_types import Document
from dolma.taggers.gopher import GopherQualityTagger
from dolma.taggers.language import FastTextLanguageTagger

# Los taggers anotan atributos en cada documento
gopher = GopherQualityTagger()
lang_tagger = FastTextLanguageTagger(model_path="lid.176.bin")

def should_keep(doc: dict) -> bool:
    text = doc["text"]
    # Filtro de idioma
    lang_pred = lang_tagger.predict(text)
    if lang_pred.label not in ("es", "pt") or lang_pred.score < 0.65:
        return False
    # Filtros Gopher
    preds = gopher.predict(text)
    attrs = {p.label: p.score for p in preds}
    if attrs.get("gopher_word_count", 0) < 50:
        return False
    if attrs.get("gopher_symbol_to_word_ratio", 1.0) > 0.1:
        return False
    return True

filtered = [doc for doc in corpus if should_keep(doc)]

9. NeMo Curator — filtrado acelerado con GPU

NeMo Curator de NVIDIA acelera el pipeline de filtrado usando RAPIDS/cuDF en GPU. La deduplicación fuzzy es ~16× más rápida que la alternativa en CPU. Útil para corpus de decenas de TB donde Datatrove en CPU es demasiado lento.

bash
pip install nemo-curator[all]

# Requiere RAPIDS (cuDF) para aceleración GPU
# En cluster con GPUs NVIDIA A100/H100
python — pipeline NeMo Curator
import nemo_curator as nc
from nemo_curator.datasets import DocumentDataset
from nemo_curator.filters import (
    FastTextLangId,
    FastTextQualityFilter,
    WordCountFilter,
    NonAlphaNumericFilter,
    RepeatedLinesByCharFilter,
)
from nemo_curator.utils.distributed_utils import get_client

# Inicializar cluster (CPU o GPU)
client = get_client(cluster_type="gpu", n_workers=4)   # GPU
# client = get_client(cluster_type="cpu", n_workers=16) # CPU

# Cargar dataset (soporta múltiples formatos)
dataset = DocumentDataset.read_json("data/raw/", add_filename=True)

# Definir pipeline de filtros
pipeline = nc.Sequential([
    # 1. Idioma
    nc.ScoreFilter(
        FastTextLangId("lid.176.bin"),
        text_field="text",
        score_field="language",
        score_type=str,
    ),

    # 2. Longitud mínima
    nc.ScoreFilter(
        WordCountFilter(min_words=50),
        text_field="text",
        score_field="word_count",
    ),

    # 3. Ratio alfanumérico
    nc.ScoreFilter(
        NonAlphaNumericFilter(max_non_alpha_numeric_ratio=0.25),
        text_field="text",
        score_field="non_alpha",
    ),

    # 4. Líneas repetidas
    nc.ScoreFilter(
        RepeatedLinesByCharFilter(max_repeated_lines_ratio=0.70),
        text_field="text",
        score_field="repeated_lines",
    ),

    # 5. Clasificador de calidad fastText
    nc.ScoreFilter(
        FastTextQualityFilter("fasttext_oh_eli5.bin", label="hq", threshold=0.018),
        text_field="text",
        score_field="quality_score",
    ),
])

# Ejecutar
filtered = pipeline(dataset)
filtered.to_json("data/filtered/", write_to_filename=True)
GPU vs CPU: Para deduplicación fuzzy de 1T+ tokens, NeMo Curator en GPU es 10-16× más rápido que datasketch en CPU. Para el filtrado de texto puro, la diferencia es menor (~2-4×). La ventaja real es la deduplicación.
